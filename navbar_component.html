Component JavaScript 
document.addEventListener('DOMContentLoaded', () => {
  const toggleBtn = document.getElementById('mobile-toggle');
  const drawerOverlay = document.getElementById('mobile-drawer');
  const navElement = document.getElementById('main-navbar');
  const mobileLinks = document.querySelectorAll('.mobile-link-item a');

  // Toggle responsive mobile menu drawer
  const toggleDrawer = () => {
    const isActive = drawerOverlay.classList.toggle('active');
    toggleBtn.classList.toggle('active');
    
    // Toggle aria state
    toggleBtn.setAttribute('aria-expanded', isActive);
    
    // Lock scrolling while mobile menu is open
    document.body.style.overflow = isActive ? 'hidden' : 'initial';
  };

  toggleBtn.addEventListener('click', toggleDrawer);

  // Close mobile drawer when selecting a navigation link
  mobileLinks.forEach(link => {
    link.addEventListener('click', () => {
      if (drawerOverlay.classList.contains('active')) {
        toggleDrawer();
      }
    });
  });

  // Shrink navbar slightly on scroll
  window.addEventListener('scroll', () => {
    if (window.scrollY > 20) {
      navElement.classList.add('shrink');
    } else {
      navElement.classList.remove('shrink');
    }
  });
});
